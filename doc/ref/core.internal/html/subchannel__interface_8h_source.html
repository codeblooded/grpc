30 // The interface for subchannels that is exposed to LB policy implementations.
31 class SubchannelInterface : public RefCounted<SubchannelInterface> {
32  public:
33  class ConnectivityStateWatcherInterface {
34  public:
35  virtual ~ConnectivityStateWatcherInterface() = default;
36 
37  // Will be invoked whenever the subchannel's connectivity state
38  // changes. There will be only one invocation of this method on a
39  // given watcher instance at any given time.
40  virtual void OnConnectivityStateChange(
41  grpc_connectivity_state new_state) = 0;
42 
43  // TODO(roth): Remove this as soon as we move to EventManager-based
44  // polling.
45  virtual grpc_pollset_set* interested_parties() = 0;
46  };

48  template <typename TraceFlagT = TraceFlag>
49  explicit SubchannelInterface(TraceFlagT* trace_flag = nullptr)
50  : RefCounted<SubchannelInterface>(trace_flag) {}
51 
52  virtual ~SubchannelInterface() = default;
53 
54  // Returns the current connectivity state of the subchannel.
55  virtual grpc_connectivity_state CheckConnectivityState() = 0;
56 
57  // Starts watching the subchannel's connectivity state.
58  // The first callback to the watcher will be delivered when the
59  // subchannel's connectivity state becomes a value other than
60  // initial_state, which may happen immediately.
61  // Subsequent callbacks will be delivered as the subchannel's state
62  // changes.
63  // The watcher will be destroyed either when the subchannel is
64  // destroyed or when CancelConnectivityStateWatch() is called.
65  // There can be only one watcher of a given subchannel. It is not
66  // valid to call this method a second time without first cancelling
67  // the previous watcher using CancelConnectivityStateWatch().
68  virtual void WatchConnectivityState(
69  grpc_connectivity_state initial_state,
70  std::unique_ptr<ConnectivityStateWatcherInterface> watcher) = 0;
71 
72  // Cancels a connectivity state watch.
73  // If the watcher has already been destroyed, this is a no-op.
74  virtual void CancelConnectivityStateWatch(
75  ConnectivityStateWatcherInterface* watcher) = 0;
76 
77  // Attempt to connect to the backend. Has no effect if already connected.
78  // If the subchannel is currently in backoff delay due to a previously
79  // failed attempt, the new connection attempt will not start until the
80  // backoff delay has elapsed.
81  virtual void AttemptToConnect() = 0;
82 
83  // Resets the subchannel's connection backoff state. If AttemptToConnect()
84  // has been called since the subchannel entered TRANSIENT_FAILURE state,
85  // starts a new connection attempt immediately; otherwise, a new connection
86  // attempt will be started as soon as AttemptToConnect() is called.
87  virtual void ResetBackoff() = 0;
88 
89  // TODO(roth): Need a better non-grpc-specific abstraction here.
90  virtual const grpc_channel_args* channel_args() = 0;
91 
92  // Allows accessing the attributes associated with the address for
93  // this subchannel.
94  virtual const ServerAddress::AttributeInterface* GetAttribute(
95  const char* key) const = 0;
96 };
