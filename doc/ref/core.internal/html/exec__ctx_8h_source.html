46 /* This exec_ctx is ready to return: either pre-populated, or cached as soon as
47  the finish_check returns true */
48 #define GRPC_EXEC_CTX_FLAG_IS_FINISHED 1
49 /* The exec_ctx's thread is (potentially) owned by a call or channel: care
50  should be given to not delete said call/channel from this exec_ctx */
51 #define GRPC_EXEC_CTX_FLAG_THREAD_RESOURCE_LOOP 2
52 /* This exec ctx was initialized by an internal thread, and should not
53  be counted by fork handlers */
54 #define GRPC_EXEC_CTX_FLAG_IS_INTERNAL_THREAD 4
55

305 class ApplicationCallbackExecCtx {
306  public:
308  ApplicationCallbackExecCtx() { Set(this, flags_); }
309 
311  ApplicationCallbackExecCtx(uintptr_t fl) : flags_(fl) { Set(this, flags_); }
312 
313  ~ApplicationCallbackExecCtx() {
314  if (reinterpret_cast<ApplicationCallbackExecCtx*>(
315  gpr_tls_get(&callback_exec_ctx_)) == this) {
316  while (head_ != nullptr) {
317  auto* f = head_;
318  head_ = f->internal_next;
319  if (f->internal_next == nullptr) {
320  tail_ = nullptr;
321  }
322  (*f->functor_run)(f, f->internal_success);
323  }
324  gpr_tls_set(&callback_exec_ctx_, reinterpret_cast<intptr_t>(nullptr));
325  if (!(GRPC_APP_CALLBACK_EXEC_CTX_FLAG_IS_INTERNAL_THREAD & flags_)) {
326  grpc_core::Fork::DecExecCtxCount();
327  }
328  } else {
329  GPR_DEBUG_ASSERT(head_ == nullptr);
330  GPR_DEBUG_ASSERT(tail_ == nullptr);
331  }
332  }
333 
334  uintptr_t Flags() { return flags_; }
335 
336  static ApplicationCallbackExecCtx* Get() {
337  return reinterpret_cast<ApplicationCallbackExecCtx*>(
338  gpr_tls_get(&callback_exec_ctx_));
339  }
340 
341  static void Set(ApplicationCallbackExecCtx* exec_ctx, uintptr_t flags) {
342  if (Get() == nullptr) {
343  if (!(GRPC_APP_CALLBACK_EXEC_CTX_FLAG_IS_INTERNAL_THREAD & flags)) {
344  grpc_core::Fork::IncExecCtxCount();
345  }
346  gpr_tls_set(&callback_exec_ctx_, reinterpret_cast<intptr_t>(exec_ctx));
347  }
348  }

350  static void Enqueue(grpc_experimental_completion_queue_functor* functor,
351  int is_success) {
352  functor->internal_success = is_success;
353  functor->internal_next = nullptr;
354 
355  ApplicationCallbackExecCtx* ctx = Get();
356 
357  if (ctx->head_ == nullptr) {
358  ctx->head_ = functor;
359  }
360  if (ctx->tail_ != nullptr) {
361  ctx->tail_->internal_next = functor;
362  }
363  ctx->tail_ = functor;
364  }
365 
367  static void GlobalInit(void) { gpr_tls_init(&callback_exec_ctx_); }
368 
370  static void GlobalShutdown(void) { gpr_tls_destroy(&callback_exec_ctx_); }
371 
372  static bool Available() { return Get() != nullptr; }
373 
374  private:
375  uintptr_t flags_{0u};
376  grpc_experimental_completion_queue_functor* head_{nullptr};
377  grpc_experimental_completion_queue_functor* tail_{nullptr};
378  GPR_TLS_CLASS_DECL(callback_exec_ctx_);
