40 // RefCount is a simple atomic ref-count.
41 //
42 // This is a C++ implementation of gpr_refcount, with inline functions. Due to
43 // inline functions, this class is significantly more efficient than
44 // gpr_refcount and should be preferred over gpr_refcount whenever possible.
45 //
46 // TODO(soheil): Remove gpr_refcount after submitting the GRFC and the paragraph
47 // above.
48 class RefCount {
49  public:
50  using Value = intptr_t;
51 
52  // `init` is the initial refcount stored in this object.
53  //
54  // TraceFlagT is defined to accept both DebugOnlyTraceFlag and TraceFlag.
55  // Note: RefCount tracing is only enabled on debug builds, even when a
56  // TraceFlag is used.
57  template <typename TraceFlagT = TraceFlag>
58  constexpr explicit RefCount(
59  Value init = 1,
60  TraceFlagT*
61 #ifndef NDEBUG
62  // Leave unnamed if NDEBUG to avoid unused parameter warning
63  trace_flag
64 #endif
65  = nullptr)
66  :
67 #ifndef NDEBUG
68  trace_flag_(trace_flag),
69 #endif
70  value_(init) {
71  }
72 
73  // Increases the ref-count by `n`.
74  void Ref(Value n = 1) {
75 #ifndef NDEBUG
76  const Value prior = value_.FetchAdd(n, MemoryOrder::RELAXED);
77  if (trace_flag_ != nullptr && trace_flag_->enabled()) {
78  gpr_log(GPR_INFO, "%s:%p ref %" PRIdPTR " -> %" PRIdPTR,
79  trace_flag_->name(), this, prior, prior + n);
80  }
81 #else
82  value_.FetchAdd(n, MemoryOrder::RELAXED);
83 #endif
84  }
85  void Ref(const DebugLocation& location, const char* reason, Value n = 1) {
86 #ifndef NDEBUG
87  const Value prior = value_.FetchAdd(n, MemoryOrder::RELAXED);
88  if (trace_flag_ != nullptr && trace_flag_->enabled()) {
89  gpr_log(GPR_INFO, "%s:%p %s:%d ref %" PRIdPTR " -> %" PRIdPTR " %s",
90  trace_flag_->name(), this, location.file(), location.line(),
91  prior, prior + n, reason);
92  }
93 #else
94  // Use conditionally-important parameters
95  (void)location;
96  (void)reason;
97 
98  value_.FetchAdd(n, MemoryOrder::RELAXED);
99 #endif
100  }
101 
102  // Similar to Ref() with an assert on the ref-count being non-zero.
103  void RefNonZero() {
104 #ifndef NDEBUG
105  const Value prior = value_.FetchAdd(1, MemoryOrder::RELAXED);
106  if (trace_flag_ != nullptr && trace_flag_->enabled()) {
107  gpr_log(GPR_INFO, "%s:%p ref %" PRIdPTR " -> %" PRIdPTR,
108  trace_flag_->name(), this, prior, prior + 1);
109  }
110  assert(prior > 0);
111 #else
112  value_.FetchAdd(1, MemoryOrder::RELAXED);
113 #endif
114  }
115  void RefNonZero(const DebugLocation& location, const char* reason) {
116 #ifndef NDEBUG
117  const Value prior = value_.FetchAdd(1, MemoryOrder::RELAXED);
118  if (trace_flag_ != nullptr && trace_flag_->enabled()) {
119  gpr_log(GPR_INFO, "%s:%p %s:%d ref %" PRIdPTR " -> %" PRIdPTR " %s",
120  trace_flag_->name(), this, location.file(), location.line(),
121  prior, prior + 1, reason);
122  }
123  assert(prior > 0);
124 #else
125  // Avoid unused-parameter warnings for debug-only parameters
126  (void)location;
127  (void)reason;
128  RefNonZero();
129 #endif
130  }
131 
132  bool RefIfNonZero() {
133 #ifndef NDEBUG
134  if (trace_flag_ != nullptr && trace_flag_->enabled()) {
135  const Value prior = get();
136  gpr_log(GPR_INFO, "%s:%p ref_if_non_zero %" PRIdPTR " -> %" PRIdPTR,
137  trace_flag_->name(), this, prior, prior + 1);
138  }
139 #endif
140  return value_.IncrementIfNonzero();
141  }
142  bool RefIfNonZero(const DebugLocation& location, const char* reason) {
143 #ifndef NDEBUG
144  if (trace_flag_ != nullptr && trace_flag_->enabled()) {
145  const Value prior = get();
146  gpr_log(GPR_INFO,
147  "%s:%p %s:%d ref_if_non_zero "
148  "%" PRIdPTR " -> %" PRIdPTR " %s",
149  trace_flag_->name(), this, location.file(), location.line(),
150  prior, prior + 1, reason);
151  }
152 #endif
153  // Avoid unused-parameter warnings for debug-only parameters
154  (void)location;
155  (void)reason;
156  return value_.IncrementIfNonzero();
157  }
158 
159  // Decrements the ref-count and returns true if the ref-count reaches 0.
160  bool Unref() {
161 #ifndef NDEBUG
162  // Grab a copy of the trace flag before the atomic change, since we
163  // can't safely access it afterwards if we're going to be freed.
164  auto* trace_flag = trace_flag_;
165 #endif
166  const Value prior = value_.FetchSub(1, MemoryOrder::ACQ_REL);
167 #ifndef NDEBUG
168  if (trace_flag != nullptr && trace_flag->enabled()) {
169  gpr_log(GPR_INFO, "%s:%p unref %" PRIdPTR " -> %" PRIdPTR,
170  trace_flag->name(), this, prior, prior - 1);
171  }
172  GPR_DEBUG_ASSERT(prior > 0);
173 #endif
174  return prior == 1;
175  }
176  bool Unref(const DebugLocation& location, const char* reason) {
177 #ifndef NDEBUG
178  // Grab a copy of the trace flag before the atomic change, since we
179  // can't safely access it afterwards if we're going to be freed.
180  auto* trace_flag = trace_flag_;
181 #endif
182  const Value prior = value_.FetchSub(1, MemoryOrder::ACQ_REL);
183 #ifndef NDEBUG
184  if (trace_flag != nullptr && trace_flag->enabled()) {
185  gpr_log(GPR_INFO, "%s:%p %s:%d unref %" PRIdPTR " -> %" PRIdPTR " %s",
186  trace_flag->name(), this, location.file(), location.line(), prior,
187  prior - 1, reason);
188  }
189  GPR_DEBUG_ASSERT(prior > 0);
190 #else
191  // Avoid unused-parameter warnings for debug-only parameters
192  (void)location;
193  (void)reason;
194 #endif
195  return prior == 1;
196  }
197 
198  private:
199  Value get() const { return value_.Load(MemoryOrder::RELAXED); }
200 
201 #ifndef NDEBUG
202  TraceFlag* trace_flag_;
203 #endif
204  Atomic<Value> value_;
205 };

236 // A base class for reference-counted objects.
237 // New objects should be created via new and start with a refcount of 1.
238 // When the refcount reaches 0, the object will be deleted via delete.
239 //
240 // If DeleteUponUnref is false, deletion will not occur when the ref
241 // count reaches 0. This is useful in cases where all existing objects
242 // must be tracked in a registry but the object's entry in the registry
243 // cannot be removed from the object's dtor due to synchronization issues.
244 // In this case, the registry can be cleaned up later by identifying
245 // entries for which RefIfNonZero() returns null.
246 //
247 // This will commonly be used by CRTP (curiously-recurring template pattern)
248 // e.g., class MyClass : public RefCounted<MyClass>
249 //
250 // Use PolymorphicRefCount and NonPolymorphicRefCount to select between
251 // different implementations of RefCounted.
252 //
253 // Note that NonPolymorphicRefCount does not support polymorphic destruction.
254 // So, use NonPolymorphicRefCount only when both of the following conditions
255 // are guaranteed to hold:
256 // (a) Child is a concrete leaf class in RefCounted<Child>, and
257 // (b) you are guaranteed to call Unref only on concrete leaf classes and not
258 // their parents.
259 //
260 // The following example is illegal, because calling Unref() will not call
261 // the dtor of Child.
262 //
263 // class Parent : public RefCounted<Parent, NonPolymorphicRefCount> {}
264 // class Child : public Parent {}
265 //
266 // Child* ch;
267 // ch->Unref();
268 //
269 template <typename Child, typename Impl = PolymorphicRefCount,
270  bool DeleteUponUnref = true>
271 class RefCounted : public Impl {
272  public:
273  // Note: Depending on the Impl used, this dtor can be implicitly virtual.
274  ~RefCounted() = default;
275 
276  RefCountedPtr<Child> Ref() GRPC_MUST_USE_RESULT {
277  IncrementRefCount();
278  return RefCountedPtr<Child>(static_cast<Child*>(this));
279  }
280 
281  RefCountedPtr<Child> Ref(const DebugLocation& location,
282  const char* reason) GRPC_MUST_USE_RESULT {
283  IncrementRefCount(location, reason);
284  return RefCountedPtr<Child>(static_cast<Child*>(this));
285  }
286 
287  // TODO(roth): Once all of our code is converted to C++ and can use
288  // RefCountedPtr<> instead of manual ref-counting, make this method
289  // private, since it will only be used by RefCountedPtr<>, which is a
290  // friend of this class.
291  void Unref() {
292  if (GPR_UNLIKELY(refs_.Unref())) {
293  internal::Delete<Child, DeleteUponUnref>(static_cast<Child*>(this));
294  }
295  }
296  void Unref(const DebugLocation& location, const char* reason) {
297  if (GPR_UNLIKELY(refs_.Unref(location, reason))) {
298  internal::Delete<Child, DeleteUponUnref>(static_cast<Child*>(this));
299  }
300  }
301 
302  RefCountedPtr<Child> RefIfNonZero() GRPC_MUST_USE_RESULT {
303  return RefCountedPtr<Child>(refs_.RefIfNonZero() ? static_cast<Child*>(this)
304  : nullptr);
305  }
306  RefCountedPtr<Child> RefIfNonZero(const DebugLocation& location,
307  const char* reason) GRPC_MUST_USE_RESULT {
308  return RefCountedPtr<Child>(refs_.RefIfNonZero(location, reason)
309  ? static_cast<Child*>(this)
310  : nullptr);
311  }
312 
313  // Not copyable nor movable.
314  RefCounted(const RefCounted&) = delete;
315  RefCounted& operator=(const RefCounted&) = delete;
316 
317  protected:
318  // TraceFlagT is defined to accept both DebugOnlyTraceFlag and TraceFlag.
319  // Note: RefCount tracing is only enabled on debug builds, even when a
320  // TraceFlag is used.
321  template <typename TraceFlagT = TraceFlag>
322  explicit RefCounted(TraceFlagT* trace_flag = nullptr,
323  intptr_t initial_refcount = 1)
324  : refs_(initial_refcount, trace_flag) {}
325 
326  private:
327  // Allow RefCountedPtr<> to access IncrementRefCount().
328  template <typename T>
329  friend class RefCountedPtr;
330 
331  void IncrementRefCount() { refs_.Ref(); }
332  void IncrementRefCount(const DebugLocation& location, const char* reason) {
333  refs_.Ref(location, reason);
334  }
335 
336  RefCount refs_;
337 };
