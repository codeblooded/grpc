146 class PemKeyCertPair {
147  public:
148  // Construct from the C struct. We steal its members and then immediately
149  // free it.
150  explicit PemKeyCertPair(grpc_ssl_pem_key_cert_pair* pair)
151  : private_key_(const_cast<char*>(pair->private_key)),
152  cert_chain_(const_cast<char*>(pair->cert_chain)) {
153  gpr_free(pair);
154  }
155 
156  // Movable.
157  PemKeyCertPair(PemKeyCertPair&& other) {
158  private_key_ = std::move(other.private_key_);
159  cert_chain_ = std::move(other.cert_chain_);
160  }
161  PemKeyCertPair& operator=(PemKeyCertPair&& other) {
162  private_key_ = std::move(other.private_key_);
163  cert_chain_ = std::move(other.cert_chain_);
164  return *this;
165  }
166 
167  // Copyable.
168  PemKeyCertPair(const PemKeyCertPair& other)
169  : private_key_(gpr_strdup(other.private_key())),
170  cert_chain_(gpr_strdup(other.cert_chain())) {}
171  PemKeyCertPair& operator=(const PemKeyCertPair& other) {
172  private_key_ = grpc_core::UniquePtr<char>(gpr_strdup(other.private_key()));
173  cert_chain_ = grpc_core::UniquePtr<char>(gpr_strdup(other.cert_chain()));
174  return *this;
175  }
176 
177  bool operator==(const PemKeyCertPair& other) const {
178  return std::strcmp(this->private_key(), other.private_key()) == 0 &&
179  std::strcmp(this->cert_chain(), other.cert_chain()) == 0;
180  }
181 
182  char* private_key() const { return private_key_.get(); }
183  char* cert_chain() const { return cert_chain_.get(); }
184 
185  private:
186  grpc_core::UniquePtr<char> private_key_;
187  grpc_core::UniquePtr<char> cert_chain_;
188 };
